**src/lib/api.ts** - ملف الاتصال بـ Groq API

```typescript
const GROQ_API_URL = 'https://api.groq.com/openai/v1/chat/completions';

export type ToneType = 'friendly' | 'professional' | 'enthusiastic' | 'diplomatic' | 'direct';

const toneInstructions = {
  friendly: {
    instruction: 'Transform this text into a warm, friendly, and approachable tone...',
    emoji: '😊'
  },
  professional: {
    instruction: 'Transform this text into a formal, professional business tone...',
    emoji: '💼'
  },
  // ... باقي النبرات
};

export async function transformText(request: TransformRequest): Promise<TransformResponse> {
  const apiKey = import.meta.env.VITE_GROQ_API_KEY;

  const response = await fetch(GROQ_API_URL, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${apiKey}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      model: 'llama-3.3-70b-versatile',
      messages: [
        { role: 'system', content: systemPrompt },
        { role: 'user', content: text }
      ],
      temperature: 0.7,
      max_tokens: 1024
    })
  });

  return data.choices[0].message.content;
}
```

#### الخطوة 3: بناء الواجهة

**src/App.tsx** - المكون الرئيسي

```tsx
function App() {
  const [inputText, setInputText] = useState('');
  const [outputText, setOutputText] = useState('');
  const [selectedTone, setSelectedTone] = useState<ToneType>('professional');
  const [isLoading, setIsLoading] = useState(false);

  const handleTransform = async () => {
    setIsLoading(true);
    const result = await transformText({
      text: inputText,
      tone: selectedTone
    });
    setOutputText(result.transformed);
    setIsLoading(false);
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-900 via-purple-900 to-slate-900">
      {/* Header */}
      <header>
        <h1>Mytone Translate</h1>
        <p>مترجم النبرة</p>
      </header>

      {/* Tone Selector */}
      <div className="flex gap-2">
        {tones.map(tone => (
          <button key={tone.id} onClick={() => setSelectedTone(tone.id)}>
            {tone.emoji} {tone.nameAr}
          </button>
        ))}
      </div>

      {/* Text Areas */}
      <div className="grid md:grid-cols-2">
        <textarea
          value={inputText}
          onChange={(e) => setInputText(e.target.value)}
          placeholder="اكتب رسالتك هنا..."
        />
        <div>{outputText || 'النتيجة ستظهر هنا'}</div>
      </div>

      {/* Transform Button */}
      <button onClick={handleTransform} disabled={isLoading}>
        {isLoading ? 'جارٍ التحويل...' : 'حوّل الآن'}
      </button>
    </div>
  );
}
```
